def scrape(source_url):
  # opening up connection, scrapping the page
  client = ureq(source_url)
  raw_html = client.read()
  client.close()

  # html parsing
  page_soup = bs(raw_html,"html.parser")

  # get only review card
  containers = page_soup.findAll("div",{"class":"styles_cardWrapper__LcCPA styles_show__HUXRb styles_reviewCard__9HxJJ"})
  
  # take the data we need from review card
  for review in containers:
      name = review.find("span", class_="typography_heading-xxs__QKBS8 typography_appearance-default__AAY17").text
      country = review.find("div", class_="typography_body-m__xgxZ_ typography_appearance-subtle__8_H2l styles_detailsIcon__Fo_ua").text
      date_time = review.find("div", class_="typography_body-m__xgxZ_ typography_appearance-subtle__8_H2l styles_datesWrapper__RCEKH").text
      # because the feedback is shown as a picture and not as text, we have to take the element we want afterwards
      stars = str(review.find("div", class_="star-rating_starRating__4rrcf star-rating_medium__iN6Ty").img).split(" ")[2]
      review_head = review.find("h2", class_="typography_heading-s__f7029 typography_appearance-default__AAY17").text
      review_body = review.find("p", class_="typography_body-l__KUYFJ typography_appearance-default__AAY17 typography_color-black__5LYEn")
      # check if body text exists
      if review_body is None:
        review_body = "None"
      else: 
        review_body = review_body.text
      
      # write data to CSV file
      write_to_json([name, country, date_time, stars, review_head, review_body])


def write_to_json(data):
    try:
        with open("reviews.csv", "a") as fopen:
            csv_writer = csv.writer(fopen)
            csv_writer.writerow(data)
    finally:
        return "Error when you try write data"

A problem

We're are getting accuracy above 90, is that right?

We're are getting accuracy above 90, is that right? The answer is no. The data is imbalance and using accuracy as evalution metric is not a good idea. For example: recall that we have 11 000 positive, 550 negative and 250 neutral reviews. A total of 12 540 reviews used in testing set (0.33 * 38 000).

Case example: If the classifier predicts all positive reviews correctly and none for negative and neutral reviews then the accuracy is : 11 000 / 12 540 = 0.877%

This means that our model will be 87.7% accurate and we will think that the model is good but in reality the model "just knows best" how the predict one class (positive reviews). To understand this let us check the f1 score below?

After seeing the f1 score it is clear that the models are just good on predicting the positives? This is the problem witht data. There are so many positive cases that the negative ones or neutral.

def calculating_f1_score(model_name, label):

  all_models = []

    model = (f1_score(y_test, clf.predict(test_x_vectors), average=None, labels=lebel))

  return all_models


def plotting_f1_scores(model):
  # plotting f1 scores
  classifiers = ["SVM","DTC","NB","LOGISITIC","ERF"]
  f1_svm, f1_dtc, f1_nb, f1_log, f1_rf = model[0], model[1],model[2],model[3],model[4]

  data = []
  for i,j,k,l,m in zip(f1_svm, f1_dtc, f1_nb, f1_log, f1_rf):
      data.append([i,j,k,l,m])
      
  x = np.arange(len(classifiers))  # the label locations
  width = 0.35  # the width of the bars

  fig, ax = plt.subplots()
  rects1 = ax.bar(x - width/2, data[0], width, label='POSITIVE')
  rects2 = ax.bar(x + width/2, data[1], width, label='NEUTRAL')
  rects3 = ax.bar(x + width, data[2], width, label='NEGATIVE')

  # Add text for labels, title and custom x-axis tick labels
  ax.set_ylabel('Scores')
  ax.set_title('F1 scores by review kind and classifier kind[Imbalanced]')
  ax.set_xticks(x)
  ax.set_xticklabels(classifiers)
  ax.legend()
  fig.show()

  #print f1 scores
  print(f'f1_svm: {f1_svm} \nf1_dtc: {f1_dtc} \nf1_nb: {f1_nb} \nf1_log: {f1_log} \nf1_rf: {f1_rf}')

if __name__ == "__main__":
